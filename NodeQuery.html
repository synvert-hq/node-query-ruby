def initialize(nql_or_ruls)
  if nql_or_ruls.is_a?(String)
    @expression = NodeQueryParser.new.parse(nql_or_ruls)
  else
    @rules = NodeRules.new(nql_or_ruls)
  end
end

def match_node?(node)
  if @expression
    @expression.match_node?(node)
  elsif @rules
    @rules.match_node?(node)
  else
    false
  end
end
