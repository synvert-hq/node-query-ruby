def match?(node, _base_node, operator = '=~')
  match =
    if NodeQuery.adapter.is_node?(node)
      @value.match(NodeQuery.adapter.get_source(node))
    else
      @value.match(node.to_s)
    end
  operator == '=~' ? match : !match
end
